function getStreak(records: Record<string, boolean>): number {
  const today = new Date();
  let streak = 0;

  for (let i = 0; i < 1000; i++) {
    const date = new Date();
    date.setDate(today.getDate() - i);
    const key = date.toISOString().split('T')[0];
    if (records[key]) {
      streak++;
    } else {
      break;
    }
  }

  return streak;
}
